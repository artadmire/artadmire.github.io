    # redux

    # 思考？

    如果让你设计一个状态管理器，你会怎么做？

    先来实现一个简单的视图更新模型：

    <!DOCTYPE html>

    <div id="box1">这是第一个盒子</div>

    var box1 = document.getElementById('box1');
    var box1 = document.getElementById('box1');

            this[key] = value;
            renderHtml(state)
        }

        box2.innerHTML = state.box2
    }

    分析上面的写法会存在什么问题？

    • 因为这个state是全局的，所以我可以在任何地方改变它。
    • state改变方式不统一，state.box1 = '也可以被改变'
    • 多人开发的过程中，一不小心，就会把数据清空，或者误修改了其他组件，副作用太大。

    因此，我们需要制定一个 state 的 “修改计划”，比如：我的修改计划是什么reducer，告诉他修改state的时候，按照我们的计划执行。

    # redux

    # 精髓

    • 通过reducer来控制对数据的修改
    • 数据不允许随意修改
    • 处理副作用
    • react-redux 的核心就是
      • redux 是全局数据
      • redux 不会触发界面更新
      • react-redux， 用 Context 的 provider 和 consumer 把 redux 的 store 串了起来。

    # createStore

    源码中，从createStore函数中，可以看到调用这个函数可以传递三个参数，返回一个store，store上有4个属性方法，这些属性方法都是在createStore函数内部定义的。

    四个方法的简单介绍：

    • createStore: 这个API 最终返回一个 store 对象，所有的功能都在这个对象上。
    • store.subscribe: 订阅state的变化，当state变化的时候会执行回调，可以多多个 subscribe， 里面的回调依次执行。
    • store.dispatch: 发出action的方法，每次dispatch 一个action ，都会执行 reducer 函数，生成新的 state， 然后执行subscribe 注册的回调。
    • store.getState: 返回当前的 state

    内部变量currentState，外部不能访问；暴露了获取方式和修改方式。这样就不会被随意修改，并且又可以轻松获取。

    function createStore(reducer, preloadedState, enhancer) {
  if (
    (typeof preloadedState === 'function' && typeof enhancer === 'function') ||
    (typeof enhancer === 'function' && typeof arguments[3] === 'function')
  ) {
    throw new Error(
      'It looks like you are passing several store enhancers to ' +
        'createStore(). This is not supported. Instead, compose them ' +
        'together to a single function. See https://redux.js.org/tutorials/fundamentals/part-4-store#creating-a-store-with-enhancers for an example.'
    )
  }
// 如果第二个参数是个函数，则赋值给enhancer，preloadedState置为undefined
  if (typeof preloadedState === 'function' && typeof enhancer === 'undefined') {
    enhancer = preloadedState
    preloadedState = undefined
  }

  if (typeof enhancer !== 'undefined') {
    if (typeof enhancer !== 'function') {
      throw new Error(
        `Expected the enhancer to be a function. Instead, received: '${kindOf(
          enhancer
        )}'`
      )
    }
    // 以上判断都是保证enhancer是一个函数 如果是函数，则说明使用了中间件，可以return了
    // enhancer就是applyMiddleware函数调用后的返回值
    // 从下面的调用情况可以看出applyMiddleware函数返回一个函数，返回的函数内部又返回了一个函数
    return enhancer(createStore)(reducer, preloadedState)
  }
  // 如果enhancer不是函数，则说明没有使用中间件，执行下面代码
  var _store;
  // 必包原理实现数据的存储
  var currentReducer = reducer; // 带有新状态的reducer函数
  var currentState = preloadedState; // 初始化状态
  var currentListeners = [];  // 存放订阅者
  var nextListeners = currentListeners; // 订阅者仓库的引用
  var isDispatching = false; // 判断是否正在改变状态
  
  // dispatch可以直接修改状态，如何修改状态，给用户自己控制
  // 通知所有订阅者
  function dispatch() {}

  // 订阅
  function subscribe() {}
  // 获取状态
  function getState() {
    if (isDispatching) {
      throw new Error('You may not call store.getState() while the reducer is executing. ' + 'The reducer has already received the state as an argument. ' + 'Pass it down from the top reducer instead of reading it from the store.');
    }
    return currentState;
  }

  function replaceReducer() {}

  var store = (_store = {
    dispatch: dispatch,
    subscribe: subscribe,
    getState: getState,
    replaceReducer: replaceReducer
  }, _store[$$observable] = observable, _store);
  return store;
}

    # dispatch

    redux状态管理模式是观察者模式，它定义了一种一对多的关系，让多个观察者对象同时监听某一个主题对象，这个主题对象的状态发生变化时就会通知所有的观察者对象，使得它们能够自动更新自己。

    观察者模式定义了对象间的一种一对多的依赖关系，当一个对象的状态发生改变时，所有依赖于它的对象都将得到通知，并自动更新。观察者模式属于行为型模式，行为型模式关注的是对象之间的通讯，观察者模式就是观察者和被观察者之间的通讯。

    所以dispatch主要做了两件事情

    • 一：改变状态
    • 二：通知订阅者
      function dispatch(action) {
// 判断是不是普通对象{}
    if (!isPlainObject(action)) {
      throw new Error('Actions must be plain objects. ' + 'Use custom middleware for async actions.');
    }
// 如果没有传递type 抛出错误
    if (typeof action.type === 'undefined') {
      throw new Error('Actions may not have an undefined "type" property. ' + 'Have you misspelled a constant?');
    }
// 正在改变状态的时候不允许调用dispatch
    if (isDispatching) {
      throw new Error('Reducers may not dispatch actions.');
    }
// 改变状态
    try {
      isDispatching = true;
      currentState = currentReducer(currentState, action);
    } finally {
      isDispatching = false;
    }

    /***
     * 通知订阅者 currentListeners 只用来查询
     * 如果在循环过程中执行subscribe/unsubscribe，有可能会报错，所以创建一个快照，
     * 使得dispath和subscribe/unsubscribe操作的不是同一个对象
     * / 
 
    var listeners = (currentListeners = nextListeners);
 
    for (var i = 0; i < listeners.length; i++) {
      var listener = listeners[i];
      listener();
    }
 
    return action;
  }

    # subscribe

    为什么需要订阅？

    到目前位置，redux还只是一个存放状态的容器而已，你可以改变，可以获取，但是什么时候发生改变并没有办法立即知道，无法实现响应式。可以把这个容器看作一个被观察者，所有的使用者看作是观察者，实现一对多的关系建立二者之间的联系。

    代码实现：

     function subscribe(listener) {
   // listener不是函数，抛出错误
      if (typeof listener !== 'function') {
        throw new Error('Expected the listener to be a function.');
      }
      // 在dispatch的时候不能store.subscribe()，也就是不能增加订阅者
      if (isDispatching) {
        throw new Error('You may not call store.subscribe() while the reducer is executing. ' + 'If you would like to be notified after the store has been updated, subscribe from a ' + 'component and invoke store.getState() in the callback to access the latest state. ' + 'See https://redux.js.org/api/store#subscribelistener for more details.');
      }
 
      var isSubscribed = true;
      // 确保两个仓库不指向同一个地址
      ensureCanMutateNextListeners();
      // 增加订阅者
      nextListeners.push(listener);
      
      // 返回一个函数，可以取消订阅
      return function unsubscribe() {
        // 如果还没有订阅，不需要执行
        if (!isSubscribed) {
          return;
        }
        // 在 dispatch的时候也不能销毁订阅者
        if (isDispatching) {
          throw new Error('You may not unsubscribe from a store listener while the reducer is executing. ' + 'See https://redux.js.org/api/store#subscribelistener for more details.');
        }
 
        isSubscribed = false;

        // 确保两个仓库不指向同一个地址
        ensureCanMutateNextListeners();

        var index = nextListeners.indexOf(listener);
        // 操作的是currentListeners的快照nextListeners
        nextListeners.splice(index, 1);

        currentListeners = null;
      };
    }

    上面代码中多次出现的一个函数，这个函数的作用：

    增加nextListener这个副本是为了避免在遍历listeners的过程中由于subscribe或者unsubscribe对listeners进行的修改而引起的某个listener被漏掉了。比如你在遍历到某个listener的时候，在这个listener中unsubscribe了一个在当前listener之前的listener，这个时候继续i ++的时候就会直接跳过当前listener的下一个listener，导致当前的listener没有被执行。

    /**
   * This makes a shallow copy of currentListeners so we can use
   * nextListeners as a temporary list while dispatching.
   *
   * 创建currentListeners的副本nextListeners，dispatching的时候使用作为临时的存储
   * 
   * This prevents any bugs around consumers calling
   * subscribe/unsubscribe in the middle of a dispatch.
   * 
   * 防止使用者在dispacth中调用订阅和取消产生的bug
   */
  function ensureCanMutateNextListeners() {
    if (nextListeners === currentListeners) {
      nextListeners = currentListeners.slice()
    }
  }

    # combineReducers

    一般在项目中有大量 state，每一个 state 的修改，都需要计划函数，如果所有的计划都写在一个 reducer 函数里面，这样就显得很繁杂，也不好理解。

    /**
 * counterReducer

//这个reducer需要接受的数据是state状态里面的state.counter！！！！！
function counterReducer(state, action) {
  switch (action.type) {
    case "INCREMENT":
      return {
        count: state.count + 1
      };
    case "DECREMENT":
      return {
        ...state,
        count: state.count - 1
      };
    default:
      return state;
  }
}
 
/**
 * infoReducer
 */
//这个reducer需要接受的数据是state状态里面的state.info！！！！！
function infoReducer(state, action) {
  switch (action.type) {
    case "SET_NAME":
      return {
        ...state,
        name: action.name
      };
    case "SET_DESCRIPTION":
      return {
        ...state,
        description: action.description
      };
    default:
      return state;
  }
}

    基于用法，需要返回一个合并的 reducer，执行完成之后，传入 state 状态和计划，可以返回一个新的 state，我们来实现以下：

    /**传入一个包含多个reducer的对象*/
function combineReducers(reducers) {
  /* 获取传入对象所有的key，reducerKeys = ['counter', 'info']*/
  const reducerKeys = Object.keys(reducers);
 
  /*返回一个合并后的新的reducer函数，接受state和action参数*/
  return function combination(state = {}, action) {
    /*初始化一个新的state*/
    const nextState = {};
 
    /*遍历循环执行所有的reducer，整合所有的state为一个新的*/
    for (let i = 0; i < reducerKeys.length; i++) {
      const key = reducerKeys[i];
      /*获取当前循环的reducer*/
      const reducer = reducers[key];
      /*传入的reducer对象的key和总的state的key一样。所以通过key值，获取reducer对应要改变的state*/
      const previousStateSingle = state[key];
      /*执行 当前循环的reducer，生成新的state*/
      const nextStateSingle = reducer(previousStateSingle, action);
 
      /**每一个新的state通过key，生成新的总的state*/
      nextState[key] = nextStateSingle;
    }
    return nextState;
  };
}

    既然知道了三个中间件，其他的细节是不是可以封装起来呢？咱们希望使用的时候是这样的：

    applyMiddleware就做了这样的事情。createStore函数中可以看出，applyMiddleware返回的是一个函数，返回值又返回了一个函数。第一个函数的参数是createStore，第二个参数的函数时reducer和preloadedState。

    function applyMiddleware(...middlewares) {
  // createStore中调用
  return (createStore) => (...args) => {
    const store = createStore(...args)
    let dispatch = () => {
      throw new Error(
        'Dispatching while constructing your middleware is not allowed. ' +
          'Other middleware would not be applied to this dispatch.'
      )
    }

    const middlewareAPI = {
      getState: store.getState,
      dispatch: (...args) => dispatch(...args),
    }
    const chain = middlewares.map((middleware) => middleware(middlewareAPI))
    dispatch = compose(...chain)(store.dispatch)

    return {
      ...store,
      dispatch,
    }
  }
}

    function compose() {

  for (var _len = arguments.length, funcs = new Array(_len), _key = 0; _key < _len; _key++) {
    funcs[_key] = arguments[_key];
  }

  if(_len === 0) {
    return function (arg) {
      return arg;
    };
  };

  if(_len === 1) {
    return funcs[0];
  }
  return funcs.reduce((a, b) => {
    return function() {
      return a(b.apply(void 0, arguments))
    }
  })
}
