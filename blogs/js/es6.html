    # 7 函数的缓存模式

    如果想要使用缓存模式来实现一个函数，我们经常会使用闭包的形式来保存数据：

    function cacheFun() {
      const cacheMap = {};
      return function (a) {
        const _cache = cacheMap[a];
        if (_cache) return _cache;
        return (cacheMap[a] = a++);
      };
    }

      const weakMap = new WeakMap();
    function cacheFun(obj) {
      const _cache = weakMap.get(obj);
      if (_cache) return _cache;
      weakMap.set(obj, "1");
      return "1";
    }
