    # Promise.all

    Promise.all([ .. ])接受一个promise数组并返回一个新的promise，

    Promise.all(['a', Promise.resolve(1), Promise.resolve(2)])
.then((res) => {
  console.log(res) // ['a', 1, 2]
})

    手写源码

    这里主要几点：

    1. 保证返回数组中值得顺序，这里使用let和resArr[i] 不使用push
    2. 保证所有得数据都返回了，这里使用count计算判断不使用resArr.length。
    3. 保证then存在，这里使用Promise.resolve(arr[i])包装一层
    Promise.myAll = function(arr) {
  return new Promise((resolve, reject) => {
    if(!Array.isArray) {
      return reject(new Error('arguments must be array'))
    }
    let count = 0;
    const len = arr.length, resArr = [];
    for(let i = 0; i < len; i++) {
      Promise.resolve(arr[i])
      .then((res) => {
        resArr[i] = res;
        count++;
        if(count === len) {
          resolve(resArr)
        }
      })
      .catch((e) => {
        reject(e)
      })
    }
  })
}

    # Promise.race

    顾名思义，Promse.race就是赛跑的意思，意思就是说，Promise.race([p1, p2, p3])里面哪个结果获得的快，就返回那个结果，不管结果本身是成功状态还是失败状态。

    let p1 = () => new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('success')
  },1000)
})

let p2 = () => new Promise((resolve, reject) => {
  setTimeout(() => {
    reject('failed')
  }, 500)
})

Promise.race([p1, p2]).then((result) => {
  console.log(result)
}).catch((error) => {
  console.log(errore((resolve, reject) => {
  setTimeout(() => {
    reject('failed')

    # Promise.allSettled

    Promise.allSettled跟Promise.all类似, 其参数接受一个Promise的数组, 返回一个新的Promise, 唯一的不同在于, 其不会进行短路, 也就是说当Promise全部处理完成后我们可以拿到每个Promise的状态, 而不管其是否处理成功.

    let p1 = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('success1')
  },1000)
})

let p2 = new Promise((resolve, reject) => {
  setTimeout(() => {
    reject('success2')
  }, 500)
})

let p3 = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve('success2')
  }, 200)
})

Promise.allSettled([p1,p2,p3])
.then((res) => {
  console.log(res)

.catch((e) => {
  console.log(e)

    手写源码

    function PromiseAllSettled(promiseArr) {
	return new Promise((resolve, reject) => {
		if(!Array.isArray(promiseArr)) {
			reject(new TypeError("arguments must be an array"))
		}
		
		const promiseLen = promiseArr.length;
		const res = [];
		let count = 0;
		for(let i=0; i<promiseLen; i++) {
			Promise.resolve(promiseArr[i])
			.then((value) => {
				res[i] = {
					status: 'fulfilled',
					value
				}
			})
			.catch((reason) => {
				res[i] = {
					status: 'rejected',
					reason
				}
			})
			.finally(() => {
				count++;
				if(count == promiseLen) {
					resolve(res)
				}
			})
		}
	})
}

    # Promise.finally

    不管成功还是失败，都会走到finally中,并且finally之后，还可以继续then。并且会将值原封不动的传递给后面的then。

    let p1 = () => {
  return  new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve('success')
    },1000)
  })
  
}
p1()
.finally(() => {
    console.log('finally')
})
.then((res) => {
  console.log(res) // success
})
// 打印顺序如下
// finally
// success

    Promise.prototype.myfinally = function(cb) {
  return this.then((res) => {
    return Promise.resolve(cb()).then(() => Promise.resolve(res))
  })
  .catch((e) => {
    return Promise.resolve(cb()).then(() => { throw e })
  })
}
