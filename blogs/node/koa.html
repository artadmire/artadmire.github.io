    # 4. createContext

    这个函数在callback中执行的，目的就是把req和res都挂在一个对象上返回。

      createContext(req, res) {
    const context = Object.create(this.context);
    const request = context.request = Object.create(this.request);
    const response = context.response = Object.create(this.response);
    context.app = request.app = response.app = this;
    context.req = request.req = response.req = req;
    context.res = request.res = response.res = res;
    request.ctx = response.ctx = context;
    request.response = response;
    response.request = request;
    context.originalUrl = request.originalUrl = req.url;
    context.state = {};
    return context;
  }

    # 6. koa-compose

    compose函数是来自 koa-compose的：

    const compose = require('koa-compose');
    1
    2

    找到这个依赖看下都做了什么？

    
function compose (middleware) {

  /**
   * @param {Object} context
   * @return {Promise}
   * @api public
   */

  return function (context, next) {
    // last called middleware #
    let index = -1
    return dispatch(0)
    function dispatch (i) {
      if (i <= index) return Promise.reject(new Error('next() called multiple times'))
      index = i
      let fn = middleware[i]
      // fnMiddleware调用的时候 没有next 所以是
      if (i === middleware.length) fn = next
      // 执行到最后一个或者fn不存在
      if (!fn) return Promise.resolve()
      try {
        // Promise.resolve包裹一层，防止app.use传入的函数不是promise
        return Promise.resolve(fn(context, dispatch.bind(null, i + 1)));
      } catch (err) {
        return Promise.reject(err)
      }
    }
  }
}

    这个函数接收两个参数，一个ctx，这个很好理解fn(context, dispatch.bind(null, i + 1))，就是context；第二个参数也是一个函数，next，执行next的时候到底执行的是那个函数呢？

    看下dispatch.bind(null, i + 1)，dispatch这个函数主要是从middleware数组中取出一个执行，这里看到i + 1，所以如果从0开始，第一个next应该是middleware[1]，也就是你第二次调用app.use传递进来的函数，如此递归调用。所以你会在初始化的时候看到奇怪的打印结果。

    其实这个执行过程就是递归调用，函数中套函数执行。先执行的函数先入栈，后执行的后入栈，出栈顺序是相反的，后入栈的先入栈，这样就形成了洋葱圈模型。

    # 应用

    
function discount (ctx, next) {
    console.log("starting discount..")
    next(ctx * 0.8) // 八折优惠
    console.log("ending discount..")
}

function express (ctx, next) {
    console.log("starting express..")
		next(ctx + 12) // 不包邮，12元运费
    console.log("ending express..")

}

function num (ctx, next) {
    console.log("starting num..")
		next(ctx * 10) // 不包邮，12元运费
    console.log("ending num..")
}

const c = compose([num, discount, express]);
console.log(c(15))

    实现compose使得可以拿到结果。

    function compose(funs) {
  if (!Array.isArray(funs)) {
    console.error('入参必须是数组')
    return; 
  }
  let result, i = 0;
  return function(ctx) {
    let dispatch = function(ctx) {
      const fn = funs[i];
      if(!fn || i === funs.length) {
        result = ctx;
        return;
      }
      i++;
      fn(ctx, dispatch.bind(null));
    }
    dispatch(ctx);
    return result;
  }
}
